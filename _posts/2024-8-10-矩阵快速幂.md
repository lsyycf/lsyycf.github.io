采用分治算法，通过位运算和运算符重载，快速求出矩阵幂

```
#include <bits/stdc++.h>
#define io cin.tie(0), cout.tie(0), ios::sync_with_stdio(false)
#define LL long long
#define ULL unsigned long long
#define EPS 1e-8
#define INF 0x7fffffff
#define SUB -INF - 1
using namespace std;
const int N = 2;
struct matrix
{
    int m[N][N];
};
matrix operator*(const matrix &a, const matrix &b)
{
    matrix c;
    memset(c.m, 0, sizeof(c.m));
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            for (int k = 0; k < N; k++)
                c.m[i][j] += a.m[i][k] * b.m[k][j];
    return c;
}
matrix pow_matrix(matrix &a, int n)
{
    matrix ans;
    memset(ans.m, 0, sizeof(ans.m));
    for (int i = 0; i < N; i++)
        ans.m[i][i] = 1;
    while (n)
    {
        if (n & 1)
            ans = ans * a;
        a = a * a;
        n = n >> 1;
    }
    return ans;
}
int main()
{
    io;
    matrix a;
    a.m[0][0] = 1, a.m[0][1] = 1, a.m[1][0] = 1, a.m[1][1] = 0;
    matrix b = pow_matrix(a, 10);
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N; j++)
            cout << b.m[i][j] << " ";
        cout << endl;
    }
    return 0;
}
```
